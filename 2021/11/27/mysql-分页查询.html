MySQL-分页查询

分页查询

一、应用场景 当要查询的条目数太多，一页显示不全

二、语法

select 查询列表 from 表 limit 【offset，】size;

注意： offset代表的是起始的条目索引，默认从0卡死 size代表的是显示的条目数

公式： 假如要显示的页数为page，每一页条目数为size

select 查询列表 from 表 limit (page-1)*size,size;

#查询平均工资最低的部门信息
SELECT
*
FROM
departments d
WHERE
d.department_id=(SELECT
department_id
FROM
employees
GROUP BY department_id
ORDER BY AVG(salary)
LIMIT 1);

#查询平均工资最低的部门信息和平均工资
SELECT
  d.*,
  avg_a.aa
FROM
  `departments` d
  JOIN
    (SELECT
      department_id,
      AVG(salary) aa
    FROM
      employees
    GROUP BY department_id
    ORDER BY AVG(salary)
    LIMIT 1) avg_a
    ON d.`department_id` =
    (SELECT
      department_id
    FROM
      employees
    GROUP BY department_id
    ORDER BY AVG(salary)
    LIMIT 1) ;
