#查询最低工资大于50号部门的最低工资的部门id和最低工资

SELECT MIN(salary)
FROM employees

SELECT manager_id,MIN(salary)
FROM employees
GROUP BY manager_id
HAVING MIN(salary)>(SELECT MIN(salary)
FROM employees

#其他部门比job id 为IT_proG部门的任意工资低的员工的工资，姓名

SELECT DISTINCT salary
FROM employees
WHERE job_id='IT_PROG'

SELECT salary,last_name,job_id
FROM employees
WHERE salary < SOME(SELECT DISTINCT salary
FROM employees
WHERE job_id='IT_PROG')
AND job_id <> 'IT_PROG';

#其他部门比job id 为IT——proG部门的所有工资低的员工的工资，姓名

AND job_id <> 'IT_PROG';

放在from后面

查询每个部门平均工资的工资等级

#平均工资
SELECT AVG(salary)
FROM employees
GROUP BY department_id;

SELECT
av_dep.平均工资,
jg.grade_level,
av_dep.部门号
FROM
(SELECT
AVG(salary) 平均工资,
department_id 部门号
FROM
employees
GROUP BY 部门号) av_dep
JOIN job_grades jg
ON av_dep.平均工资 BETWEEN jg.lowest_sal
AND jg.highest_sal ;
